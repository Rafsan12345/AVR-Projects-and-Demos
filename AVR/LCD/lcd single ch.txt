#define F_CPU 800000UL
#include <avr/io.h>
#include <util/delay.h>

void cmd(char a) {
	PORTC = a;
	PORTD = 0x20;
	_delay_ms(100);
	PORTD = 0x00;
}

void data(char a) {
	PORTC = a;
	PORTD = 0xA0;
	_delay_ms(100);
	PORTD = 0x80;
}



int main() {
	DDRC = 0xFF; DDRD = 0xE0;
	PORTC = 0x00; PORTD = 0x00;
	cmd(0x38); cmd(0x80); cmd(0x01);
	cmd(0x0E); cmd(0x06);
	
	char ascii_char = 8 + '0';
	
	data(ascii_char); 
	
	while(1);
}
