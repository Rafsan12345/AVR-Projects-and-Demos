#define F_CPU 8000000UL
#include <avr/io.h>
#include <avr/wdt.h>
#include <util/delay.h>

int main(void)
{
    DDRC |= (1 << PC0);
    wdt_enable(WDTO_2S);  // Enable WDT, 2s timeout

    while (1)
    {
        PORTC ^= (1 << PC0);
        _delay_ms(500);
        wdt_reset(); // Reset watchdog
    }
}
